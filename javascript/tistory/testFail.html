04. 결괏값을 작성하시오.

function q4() {
    const array = [10,20,30,40,50];
    return array.reduce((prev, curr) => prev += curr, 0) / array.length;
    // ⇒ reduce(function(prev, curr){ prev = prev + curr;}, 0(초기값))
    // ⇒ reduce는 prev에 초기값 또는 배열의 첫번째 자리 수가 들어가고, curr에 두번째 자리 수부터 순차적으로 들어가게 된다.
    // ⇒ 이후 prev에는 해당 식의 리턴값이 들어가게 된다.
    // ⇒ 150 / 5 = 30
}

console.log(q4());

function q6(a,b,c,d) { // a=1, b=2, c=3, d=4
    const array = [...arguments]; // 펼침연산자로 매개변수를 하나의 배열에 저장
    return array.reduceRight((prev, curr) => { // reduceRight()은 배열의 요소를 뒤에서부터 거꾸로 합한다.
        return prev += curr;
        // 4+3+2+1 = 10
    }, 0);
}

console.log(q6(1,2,3,4));

function q7() {
    const str = "taraeaea";
    return str.split('').map(word => { // 문자열을 하나씩 나눠 배열에 저장한 다음 각각의 문자를 word로 반복문 실행
        if(word !== "a") return word; // 문자가 a가 아닌 경우에만 문자 반환
    }).join(''); // 배열로 쪼개진 문자를 하나로 합침 ⇒ tree
}

function q10(arr) { // arr = [2,6,8,14]
    let flag = false; // 트리거 변수
    let n = 1;
    while(!flag) { // flag의 반대 true ⇒ false, false ⇒ true. 즉, flag = false일 떄 반복문이 실행된다.
        n++; // n = 2, 3, 4, … 
        for(let i = 1; i < arr.length; i++) {
            if((arr[0] * n) % arr[i] === 0) { // arr[0]=2, (2*n)%arr[i] 가 0일 때, flag가 true가 됨. 즉, (2*n)이 2의 배수이며, 6의 배수, 8의 배수, 14의 배수여야 한다. ⇒ 각 수의 최소공배수를 구하는 문제
                flag = true;
            }
            else {
                flag = false;
                break;
            }
        }
    }
    return arr[0] * n
}

console.log(q10([2,6,8,14]));
