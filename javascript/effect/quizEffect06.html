const quizType = document.querySelector(".quiz__type"); //타입
const quizQuestion = document.querySelector(".quiz__question"); //문제 번호, 질문
const quizSelects = document.querySelector(".quiz__selects"); //객관식 보기
const quizResert = document.querySelector(".quiz__answer .result"); //해설
const quizConfirm = document.querySelector(".quiz__answer .confilm"); //정답 확인버튼
const quizView = document.querySelector(".quiz__view"); //정답 확인버튼

let quizCount = 0;
let quizRight = 0;

//문제 출력 : 매개변수를 이용 => 편리!
const updateQuiz = (index) => {
    const questionTag = `
        <span class="number">${index + 1 + ". "}</span>
        <div class="ask">${quizInfo[index].answerAsk}</div>
    `;

    const choiceTag = `
        <label for="select1">
            <input type="radio" id="select1" class="select" name="select" value="1">
            <span class="choice">${quizInfo[index].answerChoice[1]}</span>
        </label>
        <label for="select2">
            <input type="radio" id="select2" class="select" name="select" value="2">
            <span class="choice">${quizInfo[index].answerChoice[2]}</span>
        </label>
        <label for="select3">
            <input type="radio" id="select3" class="select" name="select" value="3">
            <span class="choice">${quizInfo[index].answerChoice[3]}</span>
        </label>
        <label for="select4">
            <input type="radio" id="select4" class="select" name="select" value="4">
            <span class="choice">${quizInfo[index].answerChoice[4]}</span>
        </label>
    `;

    //문제 출력
    quizType.innerHTML = quizInfo[index].answerType;   //타입
    quizQuestion.innerHTML = questionTag;   //번호, 질문 넣어줌
    quizSelects.innerHTML = choiceTag;   //객관식 보기
    quizResert.innerHTML = quizInfo[index].answerEx;   //해설

    const quizChoice = quizSelects.querySelectorAll(".choice");
    

    for(let i=0; i<quizChoice.length; i++){
        quizChoice[i].setAttribute("onclick", "choiceSelected(this)");
    }

    //정답 확인 버튼, 해설 숨기기
    quizResert.style.display = "none";
    quizConfirm.style.display = "none";
};
updateQuiz(quizCount);

const choiceSelected = (answer) => {
    let userAnswer = answer.textContent;                          //사용자가 체크한 정답
    let currentAnswer = quizInfo[quizCount].answerResult;   //문제의 정답

    if(userAnswer == currentAnswer){
        console.log("정답");
        quizView.classList.add("like");
        quizRight++;
        console.log(quizRight);
    } else {
        console.log("오답");
        quizView.classList.add("dislike");
        quizResert.style.display = "block";
    }
    
    //정답 확인 버튼
    quizConfirm.style.display = "block";
}

const answerQuiz = () => {
    //마지막 문제
    if(quizInfo.length -1 == quizCount ){
        quizConfirm.textContent = `총 ${quizCount+1}문제 중에 ${quizRight}문제를 맞추었습니다.`
    }

    quizCount++;
    updateQuiz(quizCount);
    quizView.classList.remove("like", "dislike");
    
}
quizConfirm.addEventListener("click", answerQuiz);
